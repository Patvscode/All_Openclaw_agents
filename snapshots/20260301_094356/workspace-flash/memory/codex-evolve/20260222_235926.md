start_server() {
  mkdir -p \""'$BASE_DIR" "$SITE_DIR"
  if is_running; then
    echo "research-server already running (pid $(cat "$PID_FILE")) on :$PORT"
    return 0
  fi
  nohup "$PYTHON_BIN" -m http.server "$PORT" --bind "$HOST" --directory "$SITE_DIR" >"$LOG_FILE" 2>&1 &
  local pid=$!
  echo "$pid" > "$PID_FILE"
  sleep 0.5
  if kill -0 "$pid" 2>/dev/null; then
    echo "research-server started pid=$pid host=$HOST port=$PORT dir=$SITE_DIR"
  else
    echo "failed to start research-server; see $LOG_FILE" >&2
    return 1
  fi
}

stop_server() {
  if ! [[ -f "$PID_FILE" ]]; then
    echo "research-server not running (no pid file)"
    return 0
  fi
  local pid
  pid="$(cat "$PID_FILE" 2>/dev/null || true)"
  if [[ -z "${pid:-}" ]]; then
    rm -f "$PID_FILE"
    echo "research-server not running (empty pid file)"
    return 0
  fi
  if kill -0 "$pid" 2>/dev/null; then
    kill "$pid"
    sleep 0.5 || true
    if kill -0 "$pid" 2>/dev/null; then
      kill -9 "$pid" 2>/dev/null || true
    fi
    echo "research-server stopped pid=$pid"
  else
    echo "research-server not running (stale pid=$pid)"
  fi
  rm -f "$PID_FILE"
}

status_server() {
  if is_running; then
    local pid
    pid="$(cat "$PID_FILE")"
    local idx="$SITE_DIR/index.html"
    local ts="missing"
    [[ -f "$idx" ]] && ts="$(date -r "$idx" '"'+%Y-%m-%d %H:%M:%S %z')\"
    echo \"status=running pid="'$pid host=$HOST port=$PORT site=$SITE_DIR index_updated='"'"'$ts'"'\"
  else
    local idx=\""'$SITE_DIR/index.html"
    if [[ -f "$idx" ]]; then
      echo "status=stopped host=$HOST port=$PORT site=$SITE_DIR index_exists=yes"
    else
      echo "status=stopped host=$HOST port=$PORT site=$SITE_DIR index_exists=no"
    fi
    return 1
  fi
}

  else
    start_server
  fi
