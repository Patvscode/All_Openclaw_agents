#!/usr/bin/env python3
"""Reset persistent cron session aliases for OpenClaw agent stores.

This clears keys like `agent:coder:cron:<job-id>` from an agent's
`sessions/sessions.json` so the next cron run starts from a fresh isolated
session instead of inheriting drifted context.
"""

from __future__ import annotations

import argparse
import json
import sys
import time
from pathlib import Path

DEFAULT_STATE = Path("/home/pmello/.openclaw")
DEFAULT_JOBS = DEFAULT_STATE / "cron" / "jobs.json"


def load_json(path: Path):
    with path.open() as f:
        return json.load(f)


def save_json(path: Path, data) -> None:
    tmp = path.with_suffix(path.suffix + f".tmp.{int(time.time())}")
    with tmp.open("w") as f:
        json.dump(data, f, indent=2, ensure_ascii=False)
        f.write("\n")
    tmp.replace(path)


def load_jobs_by_name(path: Path) -> dict[str, dict]:
    raw = load_json(path)
    jobs = raw.get("jobs") if isinstance(raw, dict) else raw
    out = {}
    if isinstance(jobs, list):
        for job in jobs:
            if isinstance(job, dict) and isinstance(job.get("name"), str):
                out[job["name"]] = job
    return out


def main() -> int:
    ap = argparse.ArgumentParser(description=__doc__)
    ap.add_argument("--state-dir", type=Path, default=DEFAULT_STATE)
    ap.add_argument("--jobs", type=Path, default=DEFAULT_JOBS)
    ap.add_argument("--agent", default="coder", help="Agent id for the session store (default: coder)")
    ap.add_argument("--job-id", action="append", default=[], help="Cron job id to reset (repeatable)")
    ap.add_argument("--job-name", action="append", default=[], help="Cron job name to resolve/reset (repeatable)")
    ap.add_argument("--list", action="store_true", help="List matching alias entries instead of modifying")
    ap.add_argument("--dry-run", action="store_true", help="Show planned changes without writing")
    args = ap.parse_args()

    if not args.job_id and not args.job_name:
        ap.error("provide at least one --job-id or --job-name")

    job_ids = set(args.job_id)
    if args.job_name:
        jobs_by_name = load_jobs_by_name(args.jobs)
        for name in args.job_name:
            job = jobs_by_name.get(name)
            if not job or not isinstance(job.get("id"), str):
                print(f"warn: job name not found: {name}", file=sys.stderr)
                continue
            job_ids.add(job["id"])

    store_path = args.state_dir / "agents" / args.agent / "sessions" / "sessions.json"
    data = load_json(store_path)
