    re.compile(p, re.I)

        r'\\bplease (?:provide|clarify)\\b',
        r'\\bhow can i help\\b',

        r'\\btools? (?:listed|available)\\b',

    ]
]

COMMAND_ONLY_PREFIXES = (
    'Run exactly this command and return only the command output',
    'Run exactly this command and return only command output',
    'Run this exact command and report output',
)


@dataclass
class Finding:
    job_name: str
    job_id: str
    ts: int | None
    run_at_ms: int | None
    reason: str
    summary_excerpt: str


def load_jobs(path: Path) -> list[dict]:
    with path.open() as f:
        data = json.load(f)
    if isinstance(data, dict) and isinstance(data.get('jobs'), list):
        return data['jobs']
    if isinstance(data, list):
        return data
    raise SystemExit(f'Unsupported jobs schema in {path}')


def extract_command(message: str) -> str | None:
    if not isinstance(message, str):
        return None
    msg = message.strip()
    if not any(msg.startswith(p) for p in COMMAND_ONLY_PREFIXES):
        return None
    if ':' in msg:
        return msg.rsplit(':', 1)[-1].strip()
    return None


def iter_recent_runs(run_log: Path, max_runs: int) -> Iterable[dict]:
    if not run_log.exists():
        return []
    lines = [ln for ln in run_log.read_text(errors='replace').splitlines() if ln.strip()]
    out = []
    for ln in lines[-max_runs:]:
        try:
            out.append(json.loads(ln))
        except json.JSONDecodeError:
            continue
    return out


def classify_ok_summary(command: str, summary: str) -> str | None:
    text = (summary or '').strip()
    if not text:
        return 'ok-with-empty-summary'

    hints = COMMAND_HEADER_HINTS.get(command)
    if hints:
        if not all(h in text for h in hints):
            for pat in ASSISTANT_DRIFT_PATTERNS:
                if pat.search(text):
                    return 'assistant-chatter-for-command-only-job'
            if '?' in text and len(text) < 2500:
                return 'question-like-summary-for-command-only-job'
            return 'missing-expected-command-output-headers'
        return None

    for pat in ASSISTANT_DRIFT_PATTERNS:
        if pat.search(text):
            return 'assistant-chatter-for-command-only-job'
    if text.lower().startswith(('sure', 'hello', 'hi ', 'i can help', 'i\\'m here to help')):
        return 'assistant-greeting-for-command-only-job'
    return None


def main() -> int:

    ap.add_argument('--jobs', type=Path, default=DEFAULT_JOBS)
    ap.add_argument('--runs-dir', type=Path, default=DEFAULT_RUNS_DIR)
    ap.add_argument('--max-runs', type=int, default=10, help='Recent runs to inspect per matching job')

    jobs = load_jobs(args.jobs)

                    Finding(
                        job_name=str(job.get('name') or '<unnamed>'),
                        job_id=str(job.get('id') or ''),
                        ts=rec.get('ts') if isinstance(rec.get('ts'), int) else None,
                        run_at_ms=rec.get('runAtMs') if isinstance(rec.get('runAtMs'), int) else None,
                        reason=reason,
                        summary_excerpt=((summary or '').replace('\\n', ' '))[:240],
                    )
                )

    if args.json:
        print(json.dumps({
            'scannedCommandOnlyJobs': scanned_jobs,
            'findingCount': len(findings),
            'findings': [f.__dict__ for f in findings],
        }, indent=2))
    else:
        print(f'Scanned command-only cron jobs: {scanned_jobs}')
        if not findings:

        else:
            print(f'Suspicious successful runs: {len(findings)}')

                print(f'- {f.job_name} ({f.job_id}) reason={f.reason} runAtMs={f.run_at_ms}')
                print(f'  summary: {f.summary_excerpt}')

    return 1 if findings else 0


if __name__ == '__main__':
    sys.exit(main())
